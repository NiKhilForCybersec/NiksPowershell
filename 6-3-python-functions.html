Defining Functions

Python - Basic Function Structure 
def check_port(host: str, port: int, timeout: float = 2.0) -> dict:
    """
    Check if a port is open on a target host.
    
    Args:
        host: Target hostname or IP address
        port: Port number to check (1-65535)
        timeout: Connection timeout in seconds (default: 2.0)
    
    Returns:
        dict: Contains 'port', 'status', and optionally 'error'
    
    Example:
        >>> result = check_port("192.168.1.1", 443)
        >>> print(result['status'])
        'open'
    """
    import socket
    
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(timeout)
        result = sock.connect_ex((host, port))
        status = "open" if result == 0 else "closed"
        return {"port": port, "status": status}
    except socket.timeout:
        return {"port": port, "status": "timeout"}
    except Exception as e:
        return {"port": port, "status": "error", "error": str(e)}
    finally:
        sock.close()

# Usage
result = check_port("192.168.1.1", 443)
result = check_port("192.168.1.1", 22, timeout=5.0)

Parameter Types

Python - Function Parameters 
# Positional and keyword arguments
def create_alert(source_ip, dest_ip, severity="medium", port=None):
    alert = {
        "source": source_ip,
        "destination": dest_ip,
        "severity": severity
    }
    if port:
        alert["port"] = port
    return alert

# All valid calls:
create_alert("10.0.0.1", "192.168.1.1")
create_alert("10.0.0.1", "192.168.1.1", "high")
create_alert("10.0.0.1", "192.168.1.1", severity="critical")
create_alert("10.0.0.1", "192.168.1.1", port=443, severity="high")

# *args - variable positional arguments
def scan_ports(host, *ports):
    """Scan multiple ports on a host"""
    results = []
    for port in ports:
        results.append(check_port(host, port))
    return results

scan_ports("192.168.1.1", 22, 80, 443, 3389)

# **kwargs - variable keyword arguments
def log_event(message, **metadata):
    """Log an event with arbitrary metadata"""
    entry = {"message": message, "timestamp": datetime.now().isoformat()}
    entry.update(metadata)
    return entry

log_event("Login failed", user="admin", source_ip="10.0.0.50", attempts=3)

# Combined
def security_scan(target, *ports, timeout=2.0, **options):
    """Flexible security scanning function"""
    print(f"Scanning {target}")
    print(f"Ports: {ports}")
    print(f"Timeout: {timeout}")
    print(f"Options: {options}")

Lambda Functions

Python - Lambda (Anonymous) Functions 
# Lambda for simple operations
get_severity_score = lambda level: {"low": 1, "medium": 2, "high": 3, "critical": 4}.get(level, 0)

# Sorting with lambda
alerts = [
    {"ip": "10.0.0.1", "severity": "high"},
    {"ip": "10.0.0.2", "severity": "low"},
    {"ip": "10.0.0.3", "severity": "critical"}
]

# Sort by severity
sorted_alerts = sorted(alerts, key=lambda x: get_severity_score(x["severity"]), reverse=True)

# Filter with lambda
high_priority = list(filter(lambda x: x["severity"] in ["high", "critical"], alerts))

# Map with lambda
ips_only = list(map(lambda x: x["ip"], alerts))

Decorators

Decorators wrap functions to add functionality - perfect for logging, timing, and authentication.

Python - Security Decorators 
import functools
import time
import logging

# Timing decorator
def timer(func):
    """Measure function execution time"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} completed in {elapsed:.4f} seconds")
        return result
    return wrapper

# Logging decorator
def log_call(func):
    """Log function calls with arguments"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        logging.info(f"Calling {func.__name__} with args={args}, kwargs={kwargs}")
        try:
            result = func(*args, **kwargs)
            logging.info(f"{func.__name__} returned successfully")
            return result
        except Exception as e:
            logging.error(f"{func.__name__} raised {type(e).__name__}: {e}")
            raise
    return wrapper

# Retry decorator
def retry(max_attempts=3, delay=1):
    """Retry function on failure"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts:
                        raise
                    print(f"Attempt {attempt} failed, retrying in {delay}s...")
                    time.sleep(delay)
        return wrapper
    return decorator

# Using decorators
@timer
@log_call
@retry(max_attempts=3, delay=2)
def fetch_threat_intel(ioc):
    """Fetch threat intelligence for an IOC"""
    # API call here...
    pass

# fetch_threat_intel = timer(log_call(retry(max_attempts=3)(fetch_threat_intel)))

"""Network scanning utilities"""

import socket
from typing import List, Dict, Optional
from concurrent.futures import ThreadPoolExecutor

def get_local_ip() -> str:
    """Get the local IP address"""
    s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    try:
        s.connect(("8.8.8.8", 80))
        return s.getsockname()[0]
    finally:
        s.close()

def scan_port(host: str, port: int, timeout: float = 2.0) -> Dict:
    """Scan a single port"""
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(timeout)
        result = sock.connect_ex((host, port))
        return {"port": port, "status": "open" if result == 0 else "closed"}
    except Exception as e:
        return {"port": port, "status": "error", "error": str(e)}
    finally:
        sock.close()

def scan_ports(host: str, ports: List[int], timeout: float = 2.0, 
               threads: int = 10) -> List[Dict]:
    """Scan multiple ports using thread pool"""
    results = []
    with ThreadPoolExecutor(max_workers=threads) as executor:
        futures = {executor.submit(scan_port, host, p, timeout): p for p in ports}
        for future in futures:
            results.append(future.result())
    return sorted(results, key=lambda x: x["port"])
security_utils/hashing.py 
"""File hashing utilities"""

import hashlib
from pathlib import Path
from typing import Dict

def hash_file(filepath: str, algorithms: list = None) -> Dict[str, str]:
    """
    Calculate multiple hashes for a file.
    
    Args:
        filepath: Path to file
        algorithms: List of hash algorithms (default: md5, sha1, sha256)
    
    Returns:
        Dictionary of algorithm: hash pairs
    """
    if algorithms is None:
        algorithms = ["md5", "sha1", "sha256"]
    
    hashes = {alg: hashlib.new(alg) for alg in algorithms}
    
    with open(filepath, "rb") as f:
        while chunk := f.read(8192):
            for h in hashes.values():
                h.update(chunk)
    
    return {alg: h.hexdigest() for alg, h in hashes.items()}

def verify_hash(filepath: str, expected_hash: str, algorithm: str = "sha256") -> bool:
    """Verify file hash matches expected value"""
    actual = hash_file(filepath, [algorithm])
    return actual[algorithm].lower() == expected_hash.lower()

Key Takeaways

  • Use type hints and docstrings for maintainable code
  • *args captures variable positional args, **kwargs captures keyword args
  • Decorators add cross-cutting functionality (logging, timing, retry)
  • Organize related functions into modules with __init__.py
  • Follow consistent project structure for reusable tools
