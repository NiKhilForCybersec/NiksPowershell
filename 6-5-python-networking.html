Socket Programming

Python - Port Scanner with Sockets 
import socket
from concurrent.futures import ThreadPoolExecutor
from typing import List, Dict

def scan_port(host: str, port: int, timeout: float = 1.0) -> Dict:
    """Scan a single port"""
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(timeout)
        result = sock.connect_ex((host, port))
        
        if result == 0:
            # Try to grab banner
            try:
                sock.send(b"HEAD / HTTP/1.0\r\n\r\n")
                banner = sock.recv(1024).decode("utf-8", errors="ignore").strip()
            except:
                banner = ""
            return {"port": port, "state": "open", "banner": banner[:100]}
        return {"port": port, "state": "closed"}
    except socket.timeout:
        return {"port": port, "state": "filtered"}
    except Exception as e:
        return {"port": port, "state": "error", "error": str(e)}
    finally:
        sock.close()

def scan_host(host: str, ports: List[int], threads: int = 50) -> List[Dict]:
    """Scan multiple ports on a host using thread pool"""
    results = []
    with ThreadPoolExecutor(max_workers=threads) as executor:
        futures = {executor.submit(scan_port, host, p): p for p in ports}
        for future in futures:
            results.append(future.result())
    return sorted([r for r in results if r["state"] == "open"], key=lambda x: x["port"])

# Usage
common_ports = [21, 22, 23, 25, 53, 80, 110, 143, 443, 445, 993, 995, 3306, 3389, 5432, 8080, 8443]
open_ports = scan_host("192.168.1.1", common_ports)
for port in open_ports:
    print(f"Port {port['port']}: {port['state']} - {port.get('banner', '')}")

SSH with Paramiko

Python - SSH Connections 
import paramiko
from pathlib import Path

def ssh_execute(host: str, username: str, command: str, 
                key_path: str = None, password: str = None) -> dict:
    """Execute command via SSH"""
    client = paramiko.SSHClient()
    client.set_missing_host_key_policy(paramiko.AutoAddPolicy())
    
    try:
        # Connect with key or password
        if key_path:
            key = paramiko.RSAKey.from_private_key_file(key_path)
            client.connect(host, username=username, pkey=key)
        else:
            client.connect(host, username=username, password=password)
        
        # Execute command
        stdin, stdout, stderr = client.exec_command(command)
        
        return {
            "host": host,
            "command": command,
            "stdout": stdout.read().decode(),
            "stderr": stderr.read().decode(),
            "exit_code": stdout.channel.recv_exit_status()
        }
    finally:
        client.close()

# Execute on multiple hosts
def run_on_hosts(hosts: list, command: str, username: str, key_path: str):
    """Run command on multiple hosts"""
    results = []
    for host in hosts:
        try:
            result = ssh_execute(host, username, command, key_path=key_path)
            results.append(result)
        except Exception as e:
            results.append({"host": host, "error": str(e)})
    return results

# Collect security info from servers
hosts = ["10.0.0.10", "10.0.0.11", "10.0.0.12"]
results = run_on_hosts(hosts, "last -n 10", "admin", "~/.ssh/id_rsa")

for r in results:
    if "error" not in r:
        print(f"=== {r['host']} ===")
        print(r["stdout"])

# SFTP file transfer
def sftp_download(host, username, remote_path, local_path, key_path):
    """Download file via SFTP"""
    transport = paramiko.Transport((host, 22))
    key = paramiko.RSAKey.from_private_key_file(key_path)
    transport.connect(username=username, pkey=key)
    
    sftp = paramiko.SFTPClient.from_transport(transport)
    sftp.get(remote_path, local_path)
    sftp.close()
    transport.close()

Scapy Packet Manipulation

Python - Network Scanning with Scapy 
from scapy.all import *

# ARP scan for host discovery
def arp_scan(network: str) -> list:
    """Discover hosts on network using ARP"""
    arp = ARP(pdst=network)
    ether = Ether(dst="ff:ff:ff:ff:ff:ff")
    packet = ether/arp
    
    result = srp(packet, timeout=3, verbose=0)[0]
    
    hosts = []
    for sent, received in result:
        hosts.append({
            "ip": received.psrc,
            "mac": received.hwsrc
        })
    return hosts

# TCP SYN scan
def syn_scan(host: str, ports: list) -> list:
    """TCP SYN scan (stealth scan)"""
    open_ports = []
    
    for port in ports:
        # Send SYN packet
        syn = IP(dst=host)/TCP(dport=port, flags="S")
        response = sr1(syn, timeout=1, verbose=0)
        
        if response and response.haslayer(TCP):
            if response[TCP].flags == 0x12:  # SYN-ACK
                open_ports.append(port)
                # Send RST to close
                rst = IP(dst=host)/TCP(dport=port, flags="R")
                send(rst, verbose=0)
    
    return open_ports

# ICMP ping sweep
def ping_sweep(network: str) -> list:
    """Ping sweep a network"""
    answered, unanswered = sr(IP(dst=network)/ICMP(), timeout=2, verbose=0)
    live_hosts = [rcv.src for snd, rcv in answered]
    return live_hosts

# DNS query
def dns_lookup(domain: str, dns_server: str = "8.8.8.8") -> list:
    """Perform DNS lookup"""
    dns_query = IP(dst=dns_server)/UDP(dport=53)/DNS(rd=1, qd=DNSQR(qname=domain))
    response = sr1(dns_query, timeout=2, verbose=0)
    
    if response and response.haslayer(DNS):
        answers = []
        for i in range(response[DNS].ancount):
            answers.append(response[DNS].an[i].rdata)
        return answers
    return []

# Packet sniffing
def sniff_credentials(interface: str, count: int = 100):
    """Sniff for potential credentials in traffic"""
    def packet_callback(packet):
        if packet.haslayer(TCP) and packet.haslayer(Raw):
            payload = packet[Raw].load.decode("utf-8", errors="ignore")
            keywords = ["password", "passwd", "user", "login", "auth"]
            if any(kw in payload.lower() for kw in keywords):
                print(f"[!] Potential credentials: {packet[IP].src} -> {packet[IP].dst}")
                print(f"    Payload: {payload[:200]}")
    
    sniff(iface=interface, count=count, prn=packet_callback, 
          filter="tcp port 80 or tcp port 21 or tcp port 23")

Python-Nmap Integration

Python - Nmap Wrapper 
import nmap

def nmap_scan(target: str, ports: str = "1-1000") -> dict:
    """Perform nmap scan with service detection"""
    nm = nmap.PortScanner()
    
    # -sV: Service version detection
    # -sC: Default scripts
    # -O: OS detection (requires root)
    nm.scan(target, ports, arguments="-sV -sC")
    
    results = {"hosts": []}
    
    for host in nm.all_hosts():
        host_info = {
            "ip": host,
            "hostname": nm[host].hostname(),
            "state": nm[host].state(),
            "ports": []
        }
        
        for proto in nm[host].all_protocols():
            for port in nm[host][proto].keys():
                port_info = nm[host][proto][port]
                host_info["ports"].append({
                    "port": port,
                    "protocol": proto,
                    "state": port_info["state"],
                    "service": port_info["name"],
                    "version": port_info.get("version", ""),
                    "product": port_info.get("product", "")
                })
        
        results["hosts"].append(host_info)
    
    return results

# Vulnerability scan
def vuln_scan(target: str) -> dict:
    """Run nmap vulnerability scripts"""
    nm = nmap.PortScanner()
    nm.scan(target, arguments="--script vuln")
    return nm[target]

# Usage
results = nmap_scan("192.168.1.0/24", "22,80,443,3389")
for host in results["hosts"]:
    print(f"\n{host['ip']} ({host['hostname']})")
    for port in host["ports"]:
        if port["state"] == "open":
            print(f"  {port['port']}/{port['protocol']}: {port['service']} {port['version']}")

Key Takeaways

  • Use sockets for basic port scanning, Scapy for advanced packet work
  • requests library is essential for API integration
  • Paramiko enables SSH automation across multiple hosts
  • python-nmap wraps nmap for comprehensive scanning
  • Always handle timeouts and errors in network code
