Step 1: Asset Discovery

ğŸ”· PowerShell - Asset Discovery 
function Get-NetworkAssets {
    <#
    .SYNOPSIS
        Discovers assets via multiple methods.
    #>
    [CmdletBinding()]
    param(
        [string]$Subnet = "192.168.1",
        [int]$StartIP = 1,
        [int]$EndIP = 254,
        [switch]$UseAD
    )
    
    $Assets = @()
    
    # Method 1: Active Directory (if available)
    if ($UseAD) {
        Write-Host "[*] Querying Active Directory..." -ForegroundColor Cyan
        try {
            $ADComputers = Get-ADComputer -Filter * -Properties Name, DNSHostName, OperatingSystem, LastLogonDate
            $Assets += $ADComputers | ForEach-Object {
                [PSCustomObject]@{
                    Name = $_.Name
                    IPAddress = $null
                    Source = "ActiveDirectory"
                    OS = $_.OperatingSystem
                    LastSeen = $_.LastLogonDate
                }
            }
        } catch {
            Write-Warning "AD query failed: $_"
        }
    }
    
    # Method 2: Network scan
    Write-Host "[*] Scanning network $Subnet.$StartIP-$EndIP..." -ForegroundColor Cyan
    
    $Jobs = $StartIP..$EndIP | ForEach-Object {
        $IP = "$Subnet.$_"
        Start-Job -ScriptBlock {
            param($IP)
            if (Test-Connection $IP -Count 1 -Quiet -TimeoutSeconds 1) {
                try {
                    $DNS = [System.Net.Dns]::GetHostEntry($IP).HostName
                } catch { $DNS = $null }
                
                [PSCustomObject]@{
                    IPAddress = $IP
                    Hostname = $DNS
                    Online = $true
                }
            }
        } -ArgumentList $IP
    }
    
    $ScanResults = $Jobs | Wait-Job | Receive-Job
    $Jobs | Remove-Job
    
    $Assets += $ScanResults | Where-Object { $_ } | ForEach-Object {
        [PSCustomObject]@{
            Name = $_.Hostname
            IPAddress = $_.IPAddress
            Source = "NetworkScan"
            OS = $null
            LastSeen = Get-Date
        }
    }
    
    return $Assets
}

function Get-AssetDetails {
    <#
    .SYNOPSIS
        Collects detailed system information from a remote computer.
    #>
    [CmdletBinding()]
    param(
        [Parameter(Mandatory, ValueFromPipeline)]
        [string]$ComputerName,
        
        [PSCredential]$Credential
    )
    
    process {
        Write-Verbose "Collecting details from $ComputerName"
        
        $CimParams = @{
            ComputerName = $ComputerName
            ErrorAction = 'Stop'
        }
        if ($Credential) { $CimParams.Credential = $Credential }
        
        try {
            # Computer System
            $CS = Get-CimInstance Win32_ComputerSystem @CimParams
            
            # Operating System
            $OS = Get-CimInstance Win32_OperatingSystem @CimParams
            
            # BIOS
            $BIOS = Get-CimInstance Win32_BIOS @CimParams
            
            # Processor
            $CPU = Get-CimInstance Win32_Processor @CimParams | Select-Object -First 1
            
            # Memory
            $RAM = Get-CimInstance Win32_PhysicalMemory @CimParams | 
                Measure-Object Capacity -Sum
            
            # Disk
            $Disk = Get-CimInstance Win32_LogicalDisk @CimParams | 
                Where-Object DriveType -eq 3
            
            # Network
            $NIC = Get-CimInstance Win32_NetworkAdapterConfiguration @CimParams | 
                Where-Object IPEnabled
            
            [PSCustomObject]@{
                ComputerName = $ComputerName
                Manufacturer = $CS.Manufacturer
                Model = $CS.Model
                SerialNumber = $BIOS.SerialNumber
                OSName = $OS.Caption
                OSVersion = $OS.Version
                OSBuild = $OS.BuildNumber
                LastBoot = $OS.LastBootUpTime
                Processor = $CPU.Name
                Cores = $CPU.NumberOfCores
                RAMTotal_GB = [math]::Round($RAM.Sum / 1GB, 2)
                DiskTotal_GB = [math]::Round(($Disk.Size | Measure-Object -Sum).Sum / 1GB, 2)
                DiskFree_GB = [math]::Round(($Disk.FreeSpace | Measure-Object -Sum).Sum / 1GB, 2)
                IPAddresses = ($NIC.IPAddress | Where-Object { $_ }) -join ', '
                MACAddresses = ($NIC.MACAddress | Where-Object { $_ }) -join ', '
                CollectedAt = Get-Date
            }
        } catch {
            Write-Warning "Failed to collect from $ComputerName : $_"
            [PSCustomObject]@{
                ComputerName = $ComputerName
                Error = $_.Exception.Message
                CollectedAt = Get-Date
            }
        }
    }
}

function Get-InstalledSoftware {
    <#
    .SYNOPSIS
        Gets installed software from remote computer.
    #>
    [CmdletBinding()]
    param(
        [Parameter(Mandatory)]
        [string]$ComputerName
    )
    
    $RegistryPaths = @(
        "HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall\*"
        "HKLM:\SOFTWARE\Wow6432Node\Microsoft\Windows\CurrentVersion\Uninstall\*"
    )
    
    try {
        Invoke-Command -ComputerName $ComputerName -ScriptBlock {
            param($Paths)
            foreach ($Path in $Paths) {
                Get-ItemProperty $Path -ErrorAction SilentlyContinue | 
                    Where-Object DisplayName |
                    Select-Object DisplayName, DisplayVersion, Publisher, InstallDate
            }
        } -ArgumentList (,$RegistryPaths) | ForEach-Object {
            [PSCustomObject]@{
                ComputerName = $ComputerName
                Software = $_.DisplayName
                Version = $_.DisplayVersion
                Publisher = $_.Publisher
                InstallDate = $_.InstallDate
            }
        }
    } catch {
        Write-Warning "Failed to get software from $ComputerName"
    }
}

# Security check: Find specific software
function Find-SecuritySoftware {
    param([string]$ComputerName)
    
    $SecurityProducts = @("*Defender*", "*Norton*", "*McAfee*", "*Symantec*", "*CrowdStrike*")
    
    $Software = Get-InstalledSoftware $ComputerName
    $Software | Where-Object {
        $Name = $_.Software
        $SecurityProducts | Where-Object { $Name -like $_ }
    }
}

function Invoke-AssetInventory {
    <#
    .SYNOPSIS
        Complete asset inventory collection.
    #>
    [CmdletBinding()]
    param(
        [string[]]$ComputerNames,
        [string]$Subnet,
        [string]$OutputPath = ".\Inventory_$(Get-Date -Format 'yyyyMMdd')",
        [switch]$IncludeSoftware
    )
    
    New-Item $OutputPath -ItemType Directory -Force | Out-Null
    
    # Get target list
    if ($Subnet) {
        Write-Host "[*] Discovering assets on $Subnet..." -ForegroundColor Cyan
        $Discovered = Get-NetworkAssets -Subnet $Subnet
        $ComputerNames = $Discovered | Where-Object IPAddress | Select-Object -ExpandProperty IPAddress
    }
    
    Write-Host "[*] Collecting details from $($ComputerNames.Count) assets..." -ForegroundColor Cyan
    
    # Collect hardware info
    $HardwareInventory = $ComputerNames | ForEach-Object {
        Write-Host "    Processing $_"
        Get-AssetDetails -ComputerName $_
    }
    
    # Export hardware
    $HardwareInventory | Export-Csv "$OutputPath\hardware.csv" -NoTypeInformation
    $HardwareInventory | ConvertTo-Json -Depth 3 | Out-File "$OutputPath\hardware.json"
    
    # Collect software if requested
    if ($IncludeSoftware) {
        Write-Host "[*] Collecting software inventory..." -ForegroundColor Cyan
        $SoftwareInventory = $ComputerNames | ForEach-Object {
            Get-InstalledSoftware -ComputerName $_
        }
        $SoftwareInventory | Export-Csv "$OutputPath\software.csv" -NoTypeInformation
    }
    
    # Generate summary
    $Summary = [PSCustomObject]@{
        ScanDate = Get-Date
        TotalAssets = $HardwareInventory.Count
        SuccessfulScans = ($HardwareInventory | Where-Object { -not $_.Error }).Count
        FailedScans = ($HardwareInventory | Where-Object Error).Count
        OSBreakdown = $HardwareInventory | Group-Object OSName | Select-Object Name, Count
    }
    
    $Summary | ConvertTo-Json -Depth 3 | Out-File "$OutputPath\summary.json"
    
    Write-Host "`n[+] Inventory complete: $OutputPath" -ForegroundColor Green
    Write-Host "    Total: $($Summary.TotalAssets) | Success: $($Summary.SuccessfulScans) | Failed: $($Summary.FailedScans)"
    
    return $HardwareInventory
}

# Usage
$Inventory = Invoke-AssetInventory -Subnet "192.168.1" -IncludeSoftware

Best Practices: Asset Inventory
  • Schedule regular scans: Weekly or monthly full inventory
  • Use parallelization: Jobs or runspaces for large networks
  • Handle failures gracefully: Don't stop on single failures
  • Track changes: Compare current to previous inventory
  • Secure credentials: Use credential manager for stored creds
  • Export multiple formats: CSV for Excel, JSON for automation

Key Takeaways

  • Combine AD, network scan, and manual input for complete coverage
  • Use CIM/WMI for hardware details (Win32_* classes)
  • Query registry for installed software
  • Export to CSV and JSON for flexibility
  • Track security software presence for compliance
  • Great project for demonstrating automation skills
