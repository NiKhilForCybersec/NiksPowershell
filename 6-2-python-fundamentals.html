Variables and Data Types

Python is dynamically typed - you don't declare types explicitly. The interpreter infers types at runtime.

Python - Basic Data Types

# None - represents absence of value

Python - Lists for Security Data 
# List creation
suspicious_ips = ["10.0.0.50", "192.168.1.200", "10.0.0.75"]
ports_to_scan = [22, 80, 443, 3389, 445, 139]

first_two = suspicious_ips[0:2]   # ["10.0.0.50", "192.168.1.200"]

all_ips = suspicious_ips + ["10.0.0.99"]  # Concatenate

# Sorting
ports_to_scan.sort()                      # In-place sort
sorted_ports = sorted(ports_to_scan)      # Return new sorted list

Sets - Unique Values

Python - Sets for Deduplication 
# Sets contain unique values only
all_ips = ["10.0.0.1", "10.0.0.2", "10.0.0.1", "10.0.0.3", "10.0.0.2"]
unique_ips = set(all_ips)  # {"10.0.0.1", "10.0.0.2", "10.0.0.3"}

# Set operations (great for comparing IOC lists)
blocklist_a = {"10.0.0.1", "10.0.0.2", "10.0.0.3"}
blocklist_b = {"10.0.0.2", "10.0.0.3", "10.0.0.4"}

common = blocklist_a & blocklist_b         # Intersection: {"10.0.0.2", "10.0.0.3"}
all_blocked = blocklist_a | blocklist_b    # Union: {"10.0.0.1", "10.0.0.2", "10.0.0.3", "10.0.0.4"}
only_in_a = blocklist_a - blocklist_b      # Difference: {"10.0.0.1"}
exclusive = blocklist_a ^ blocklist_b      # Symmetric diff: {"10.0.0.1", "10.0.0.4"}

Python - Conditional Logic 
# Basic if/elif/else
severity_score = 85

if severity_score >= 90:
    priority = "critical"
    notify_soc = True
elif severity_score >= 70:
    priority = "high"
    notify_soc = True

    priority = "medium"

    print("Alert: High priority incident detected!")

# Match statement (Python 3.10+)
event_type = "login_failure"

match event_type:
    case "login_failure":
        action = "check_bruteforce"
    case "file_download":
        action = "scan_file"
    case "process_execution":
        action = "check_hash"
    case _:
        action = "log_event"

Loops

Python - Loops for Security Automation 
# For loop - iterate over collection
suspicious_ips = ["10.0.0.50", "192.168.1.200", "10.0.0.75"]

for ip in suspicious_ips:
    print(f"Scanning {ip}...")

# For loop with index
for index, ip in enumerate(suspicious_ips):
    print(f"{index + 1}. {ip}")

# Range-based loop
for port in range(1, 1025):  # Ports 1-1024
    # scan port...
    pass

# While loop
retry_count = 0
max_retries = 3
connected = False

while not connected and retry_count < max_retries:
    print(f"Connection attempt {retry_count + 1}...")
    # try to connect...
    retry_count += 1

# Loop control
for ip in suspicious_ips:
    if ip.startswith("192.168"):
        continue  # Skip internal IPs
    if ip == "0.0.0.0":
        break     # Stop processing
    print(f"Processing external IP: {ip}")

# Else clause (runs if loop completes without break)
for ip in suspicious_ips:
    if ip == "malware.bad.com":
        print("Found malware C2!")
        break
else:
    print("No known malware C2 found")

List Comprehensions

List comprehensions provide a concise way to create lists. Master these for efficient Python scripting.

Python - List Comprehensions 
# Basic comprehension
ports = [22, 80, 443, 8080, 8443]
high_ports = [p for p in ports if p > 1024]  # [8080, 8443]

# Transform values
port_strings = [str(p) for p in ports]  # ['22', '80', '443', ...]

# Multiple conditions
log_lines = [
    "ERROR Failed login",
    "INFO User logged in",
    "ERROR Connection timeout",
    "WARNING Disk space low"
]
errors = [line for line in log_lines if "ERROR" in line]

# Nested comprehension
matrix = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
flattened = [num for row in matrix for num in row]  # [1,2,3,4,5,6,7,8,9]

# Dictionary comprehension
port_status = {port: "open" for port in ports if port < 1000}
# {22: 'open', 80: 'open', 443: 'open'}

# Set comprehension
unique_levels = {line.split()[0] for line in log_lines}
# {'ERROR', 'INFO', 'WARNING'}

Exception Handling

Python - Robust Error Handling 
import socket

def scan_port(host, port):
    """Scan a single port with proper error handling"""
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(2)
        result = sock.connect_ex((host, port))
        
        if result == 0:
            return {"port": port, "status": "open"}
        else:
            return {"port": port, "status": "closed"}
            
    except socket.timeout:
        return {"port": port, "status": "timeout"}
    except socket.gaierror as e:
        return {"port": port, "status": "dns_error", "error": str(e)}
    except PermissionError:
        return {"port": port, "status": "permission_denied"}
    except Exception as e:
        return {"port": port, "status": "error", "error": str(e)}
    finally:
        sock.close()  # Always close socket

# Raising exceptions
def validate_ip(ip):
    """Validate IP address format"""
    parts = ip.split(".")
    if len(parts) != 4:
        raise ValueError(f"Invalid IP format: {ip}")
    for part in parts:
        if not 0 <= int(part) <= 255:
            raise ValueError(f"Invalid octet value: {part}")
    return True

Best Practice

Catch specific exceptions, not just generic Exception. This helps identify the actual problem and handle each case appropriately.

Key Takeaways

  • Python is dynamically typed: str, int, float, bool, None
  • Lists are ordered and mutable; dicts are key-value pairs; sets are unique
  • Use f-strings for formatting: f"IP: {ip}"
  • List comprehensions are concise and Pythonic: [x for x in list if condition]
  • Always handle exceptions properly for robust scripts
