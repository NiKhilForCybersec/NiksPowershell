Security-Focused Cmdlets

üî∑ PowerShell - Security Quick Reference 
# Process Investigation
Get-Process                               # All processes
Get-Process | Sort CPU -Desc | Select -First 10  # Top CPU
Get-CimInstance Win32_Process | Select Name, CommandLine

# Network Connections
Get-NetTCPConnection -State Established
Get-NetTCPConnection | Where RemotePort -eq 443

# Event Logs
Get-WinEvent -LogName Security -MaxEvents 100
Get-WinEvent -FilterHashtable @{LogName='Security';Id=4625}  # Failed logins

# Services
Get-Service | Where Status -eq 'Running'
Get-Service | Where StartType -eq 'Automatic'

# Scheduled Tasks
Get-ScheduledTask | Where State -eq 'Ready'

# Active Directory (requires AD module)
Get-ADUser -Filter * -Properties LastLogonDate
Get-ADUser -Filter {Enabled -eq $false}
Get-ADGroupMember -Identity "Domain Admins"

# File Hashing
Get-FileHash file.exe -Algorithm SHA256
Get-ChildItem *.exe | Get-FileHash

Security One-Liners

üêß Bash - Security Quick Reference 
# Failed SSH logins - top IPs
grep "Failed password" /var/log/auth.log | \
    grep -oE '[0-9]+\.[0-9]+\.[0-9]+\.[0-9]+' | \
    sort | uniq -c | sort -rn | head

# Listening ports
ss -tulnp | grep LISTEN
netstat -tulnp 2>/dev/null | grep LISTEN

# Established connections
ss -tunap | grep ESTABLISHED

# Find SUID binaries
find / -perm -4000 -type f 2>/dev/null

# World-writable files
find / -type f -perm -o+w 2>/dev/null

# Recently modified files (last 24h)
find /var/log -mtime -1 -type f

# Users with UID 0 (root)
awk -F: '$3 == 0 {print $1}' /etc/passwd

# Cron jobs - all users
for user in $(cut -d: -f1 /etc/passwd); do
    crontab -u $user -l 2>/dev/null
done

# Process tree
ps auxf

# File hash
sha256sum /usr/bin/bash

# Open files by process
lsof -p $PID

# Monitor file changes in real-time
inotifywait -m /path/to/watch

Batch Security Commands

üì¶ Batch - Security Quick Reference 
@echo off
REM System Information
hostname
systeminfo | findstr /i "OS Name OS Version"

REM Network Status
ipconfig /all
netstat -an | findstr LISTENING

whoami /all
net user
net localgroup administrators

REM Running Processes

wmic process get name,processid,commandline

REM Services
sc query state= all
net start

REM Scheduled Tasks
schtasks /query /fo LIST /v

REM Startup Programs
wmic startup get caption,command

REM File Search
dir /s /b C:\*.exe 2>nul | findstr /i "suspicious"
